00110             except:

00130     def getOpener(self):
00131         if self.login=="" or self.password=="":
00132             opener = urllib2.build_opener(MultipartPostHandler.MultipartPostHandler)
00133             self.opener=opener
00134             return

00137         f = opener.open('http://%s%s' %(self.host, self.loginUrl))
00138         s=BeautifulSoup(f)
00139         form=s.form
00140         action=form["action"]
00141         inp=form.find("input", attrs={"name":"wpLoginToken"})
