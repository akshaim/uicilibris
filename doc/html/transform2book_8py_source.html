00034 # handle itemizations/enumerations
00035 def transform_itemenums(string, state):
00036     preamble = ""   # for enumeration/itemize environment commands
00037 
00038     # handle itemizing/enumerations
00039     p = re.compile("^([\*\#]+).*$")
00040     m = p.match(string)
00041     if (m == None):
00042         my_enum_item_level = ""
00043     else:
00044         my_enum_item_level = m.group(1)
00045 
00046     # trivial: old level = new level
00047     if (my_enum_item_level == state.enum_item_level):
00048         pass
00049     else:
00050         # find common part
00051         common = -1 
00052         while (len(state.enum_item_level) > common + 1) and \
00053                 (len(my_enum_item_level) > common + 1) and \
00054                 (state.enum_item_level[common+1] == my_enum_item_level[common+1]):
00055             common = common + 1
00056 
00057         # close enum_item_level environments from back to front
00058         for i in range(len(state.enum_item_level)-1, common, -1):
00059             if (state.enum_item_level[i] == "*"):
00060                 preamble = preamble + "\\end{itemize}\n"
00061             elif (state.enum_item_level[i] == "#"):
00062                 preamble = preamble + "\\end{enumerate}\n"
00063         
00064         # open my_enum_item_level environments from front to back
00065         for i in range(common+1, len(my_enum_item_level)):
00066             if (my_enum_item_level[i] == "*"):
00067                 preamble = preamble + "\\begin{itemize}\n"
00068             elif (my_enum_item_level[i] == "#"):
00069                 preamble = preamble + "\\begin{enumerate}\n"
00070     state.enum_item_level = my_enum_item_level
00071     
00072     # now, substitute item markers
00073     p = re.compile("^([\*\#]+)(.*)$")
00074     _string = p.sub(r"  \\item\2", string)
00075     string = preamble + _string
00076  
00077     return string

00162 def transform_tables(string, state, report=None):
00163     ### begin of included mediawiki page
00164     p = re.compile(r"^<!-- uicilibris: begin '(.*)' -->$")
00165     m=p.match(string)
00166     if m:
00167         state.currentPage=m.group(1)
00168         return string
00169     ### end of included mediawiki page
00170     p = re.compile(r"^<!-- uicilibris: end .* -->$")
00171     if p.match(string):
00172         return flushTables(string, state, report)
00173     ### table header
00174     p = re.compile("^({\|)(.*)$")
00175     m=p.match(string)
00176     if m:
00177         t=tableState()
00178         # recognition of latex="some columns format"
00179         p1 = re.compile("^{\|.*latex=&amp;quot;([^&]+)&amp;quot;")
00180         m1=p1.match(string)
00181         if m1:
00182             t.setColFormat(m1.group(1)) 
00183         state.tableStack.append(t)
00184         state.allTables.append(t)
00185         return t.latexHeader()
00186     ### table new line
00187     p = re.compile(r"^\|-+\s*$")
00188     m=p.match(string)
00189     if m:
00190         t=topOfStack(state.tableStack)
00191         if t!=None:
00192             t.addLine()
00193             return t.latexLine()
00194         else:
00195             reportErr("new table line, with no table (%s)" %state.currentPage, report)
00196             state.tableErrors.append("line")
00197             return string
00198     ### table end
00199     p = re.compile(r"^\|}\s*$")
00200     m=p.match(string)
00201     if m:
00202         if len(state.tableStack)>0:
00203             t=state.tableStack.pop()
00204             t.close()
00205             return t.latexFooter()
00206         else:
00207             reportErr("table footer with no table (%s)" %state.currentPage, report)
00208             state.tableErrors.append("footer")
00209             return string
00210     ### table cell
00211     p = re.compile(r"^\|([^\|].*)?$")
00212     m=p.match(string)
00213     if m:
00214         t=topOfStack(state.tableStack)
00215         if t != None:
00216             t.addCell()
00217             return t.latexCell(m.group(1))
00218         else:
00219             reportErr("table cell with no table (%s). This can be due to a double closing curly brace '}}' in a mathematic formula, which is inside a template. To avoid corrupting the model, replace '}}' by '} }' in the math formula." %state.currentPage, report)
00220             state.tableErrors.append("cell")
00221             return string
00222     p = re.compile(r"^!([^!].*)?$")
00223     m=p.match(string)
00224     if m:
00225         t=topOfStack(state.tableStack)
00226         if t != None:
00227             t.addCell()
00228             return t.latexCell("\\textbf{%s}" %m.group(1))
00229         else:
00230             reportErr("table highlighted cell with no table (%s)" %state.currentPage, report)
00231             state.tableErrors.append("highlighted cell")
00232             return string
00233     # default case: do nothing
00234     return string
