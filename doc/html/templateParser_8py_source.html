00051         return "templateParser(%s, ...)" %self.ident

00053     ##
00054     # 
00055     #         makes every substitution in a given text and returns the result
00056     #         @param text the given text
00057     #         @return the result with every substitution done
00058     #         
00059     def sub(self, text):
00060         return self.pattern.sub(self.parseFunc, text)
00061 
00062     ##
00063     # 
00064     #         creates a parse function
00065     #         @param fun a function getting the dictionary of parameters of a template
00066     #         @return a function whose profile is: input parameter = a match expression, result = a string
00067     #         
00068     def pfMaker(self,fun):
00069         def pf(m):
00070             params=m.group(1)
00071             params=params[1:].split("|")
00072             dic={}
00073             i=0
00074             for p in params:
00075                 m=re.match("([\S]+)=(.*)",p)
00076                 if m:
00077                     dic[m.group(1)]=m.group(2)
00078                 else:
00079                     dic[i]=p
00080                     i+=1
00081             try:
00082                 return fun(dic)
00083             except:
00084                 exctype, excval = sys.exc_info()[:2]
00085                 exc=re.findall(".*\.(.*)'.*", "%s" %exctype)[0]
00086                 pass
00087             return "{\\textbf{%s: %s}; could not translate the template '%s'}" %(exc, excval, self.ident)
00088         return pf
