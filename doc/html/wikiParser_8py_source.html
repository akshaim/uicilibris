00160     ##
00161     # 
00162     #         join lines ending with unescaped percent signs,
00163     #         unless inside codemode or nowiki mode
00164     #         @param lines a list of text lines
00165     #         
00166     def joinLines(self, lines):
00167         nowikimode = False
00168         codemode = False
00169         r = []  # result array
00170         s = ''  # new line
00171         for _l in lines:
00172             if self.encoding:
00173                 _l=_l.encode(self.encoding)
00174             (_,nowikimode) = get_nowikimode(_l, nowikimode)
00175             if not nowikimode:
00176                 (_,codemode) = get_codemode(_l, codemode)
00177 
00178             if not codemode:
00179                 l = _l.rstrip() # return chars are kept only for <code>
00180             else:
00181                 l = _l
00182 
00183             #restore markups like <math></math>, etc.
00184             for m in ('math', 'code', 'ref'):
00185                 l=re.sub('&amp;lt;(/?)'+m+'&gt;', r'<\1'+m+'>', l)
00186             l=re.sub('"&amp;lt;references */&gt;','<references />',l)
00187 
00188             if (not self.oldWiki2) and l=="" and r and r[-1]!="\n\n" :
00189                 l="\n\n" # simplify and keep the return codes for LaTeX
00190             if not (nowikimode or codemode) and (len(l) > 1) and (l[-1] == "%") and (l[-2] != "\\"):
00191                 s = s + l[:-1]
00192             elif not (nowikimode or codemode) and (l == "%"):
00193                 s = s + l[:-1]
00194             else: # when l=="" or codemode, or nowikimode, or ...
00195                 # output a line
00196                 s = s + l
00197                 if (not self.oldWiki2) and len(s)>0:
00198                     r.append(s+"\n")
00199                 else:
00200                     r.append(s)
00201                 s = ''
00202                 
00203 
00204         return r
