00041 # handle itemizations/enumerations
00042 def transform_itemenums(string, state):
00043     preamble = ""   # for enumeration/itemize environment commands
00044 
00045     # handle itemizing/enumerations
00046     p = re.compile("^([\*\#]+).*$")
00047     m = p.match(string)
00048     if (m == None):
00049         my_enum_item_level = ""
00050     else:
00051         my_enum_item_level = m.group(1)
00052 
00053     # trivial: old level = new level
00054     if (my_enum_item_level == state.enum_item_level):
00055         pass
00056     else:
00057         # find common part
00058         common = -1 
00059         while (len(state.enum_item_level) > common + 1) and \
00060                 (len(my_enum_item_level) > common + 1) and \
00061                 (state.enum_item_level[common+1] == my_enum_item_level[common+1]):
00062             common = common + 1
00063 
00064         # close enum_item_level environments from back to front
00065         for i in range(len(state.enum_item_level)-1, common, -1):
00066             if (state.enum_item_level[i] == "*"):
00067                 preamble = preamble + "\\end{itemize}\n"
00068             elif (state.enum_item_level[i] == "#"):
00069                 preamble = preamble + "\\end{enumerate}\n"
00070         
00071         # open my_enum_item_level environments from front to back
00072         for i in range(common+1, len(my_enum_item_level)):
00073             if (my_enum_item_level[i] == "*"):
00074                 preamble = preamble + "\\begin{itemize}\n"
00075             elif (my_enum_item_level[i] == "#"):
00076                 preamble = preamble + "\\begin{enumerate}\n"
00077     state.enum_item_level = my_enum_item_level
00078     
00079     # now, substitute item markers
00080     p = re.compile("^([\*\#]+)(.*)$")
00081     _string = p.sub(r"  \\item \2", string)
00082     string = preamble + _string
00083  
00084     return string

00206 def _transform_mini_parser(character, replacement, string):
00207     # implemented as a state-machine
00208     output, typewriter = [], []
00209     seen_at, seen_escape = False, False
00210     for char in string:
00211         if seen_escape:
00212             if char == character:
00213                 output.append(character)
00214             else:
00215                 output.append('\\' + char)
00216             seen_escape = False
00217         elif char == "\\":
00218             seen_escape = True
00219         elif char == character:
00220             if seen_at:
00221                 seen_at = False
00222                 output, typewriter = typewriter, output
00223                 output.append('\\'+replacement+'{')
00224                 output += typewriter
00225                 output.append('}')
00226                 typewriter = []
00227             else:
00228                 seen_at = True
00229                 output, typewriter = typewriter, output
00230         else:
00231             output.append(char)
00232     if seen_at:
00233         output, typewriter = typewriter, output
00234         output.append(character)
00235         output += typewriter
00236     return "".join(output)
