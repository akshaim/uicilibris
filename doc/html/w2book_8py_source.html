00132     # 
00133     #         @return a sanitized output: get rid of &amp; etc.
00134     #         processes the comments left by previous works;
00135     #         processes underscores in lines where there is no maths
00136     #         turns lines with math-only contents to out-of text formulas
00137     #         takes in account url-like words

00139     def sanitize(self, s):
00140         s=s.replace("&quot;","\"")
00141         s=s.replace("&amp;lt;math&gt;", "$")
00142         s=s.replace("&amp;lt;/math&gt;", "$")
00143         s=s.replace("&amp;lt;code&gt;", "\\texttt{")
00144         s=s.replace("&amp;lt;/code&gt;", "}")
00145         s=s.replace("&amp;lt;ref&gt;", "\\footnote{")
00146         s=s.replace("&amp;lt;/ref&gt;", " }")
00147         s=s.replace("&amp;lt;references /&gt;", "")
00148         s=s.replace("&amp;lt;", "<")
00149         s=s.replace("&amp;gt;", ">")
00150         s=s.replace("&gt;", ">")
00151         s=s.replace("&amp;amp;","\\&")
00152         s=s.replace("%", "\\%{}")
00153         s=re.sub(r"\\comment\{(.*)\}",r"%% \1",s)
00154         # some mediawiki instances do not provide the same results,
00155         # so here is a second pass to translate special markup
00156         s=s.replace("<math>", "$")
00157         s=s.replace("</math>", "$")
00158         s=s.replace("<code>", "\\texttt{")
00159         s=s.replace("</code>", "}")
00160         s=s.replace("<ref>", "\\footnote{")
00161         s=s.replace("</ref>", " }")
00162         s=s.replace("<references />", "")
00163         if "$" not in s:
00164             s=s.replace("_","\\_")
00165         if len(s)>2 and s[0]=="$" and s[-1]=="$":
00166             s="$%s$" %s
00167         # hyperlinks with an anchor
00168         olds=s
00169         s=re.sub(r"\[(http://[^ \}]+) +([^\}]+)]", r"\\href{\1 }{\mbox{\2} }",s)
00170         # http urls alone
00171         if olds==s: # avoid processing links twice
00172             s=re.sub(r"(http://[^ \}]+)", r"\\href{\1 }{\mbox{\1} }",s)
00173         return s

00207                     result.append(transform(line, self.state, report))
